CREATE TABLE `task_list` (
  `TASK_ID` int(11) NOT NULL AUTO_INCREMENT,
  `TASK_NAME` varchar(255) DEFAULT NULL,
  `GROUP_ID` int(11) DEFAULT NULL,
  `CREAT_TIME_STMP` timestamp NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  `UPD_TIME_STMP` timestamp NOT NULL DEFAULT '0000-00-00 00:00:00',
  `STATUS` int(11) DEFAULT '0',
  PRIMARY KEY (`TASK_ID`)
) ENGINE=InnoDB AUTO_INCREMENT=43 DEFAULT CHARSET=utf8;




CREATE TABLE `task_dur` (
  `DUR_ID` int(11) NOT NULL AUTO_INCREMENT,
  `TASK_ID` int(11) DEFAULT NULL,
  `START_TIMESTAMP` bigint(20) DEFAULT NULL,
  `END_TIMESTAMP` bigint(20) DEFAULT NULL,
  `DURATION` bigint(20) DEFAULT NULL,
  `CREAT_TIMESTAMP` timestamp NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  `UPD_TIMESTAMP` timestamp NOT NULL DEFAULT '0000-00-00 00:00:00',
  `CURR_DATE` date DEFAULT NULL,
  PRIMARY KEY (`DUR_ID`),
  KEY `FK_TASK_ID_idx` (`TASK_ID`),
  CONSTRAINT `FK_TASK_ID` FOREIGN KEY (`TASK_ID`) REFERENCES `task_list` (`TASK_ID`) ON DELETE NO ACTION ON UPDATE NO ACTION
) ENGINE=InnoDB AUTO_INCREMENT=39 DEFAULT CHARSET=utf8;
